## 一、实验设计概述

### 1.1 研究背景与目标

根据论文《多目标约束下的制冷剂分子设计：从代际演化到 AI 驱动范式》附录方向一的要求，本实验方案旨在构建一个**多尺度物理信息嵌入的制冷剂分子数据库**，为基于生成式 AI 的多目标分子设计优化提供高质量数据基础。

**核心目标**：
1. 整合量子化学、分子动力学、状态方程三个尺度的物理信息
2. 建立标准化的数据收集与预处理流程
3. 支持多目标约束下的生成模型训练与优化

### 1.2 多尺度数据框架

根据论文 6.3 节"状态方程与量子计算融合"的指导，数据收集涵盖以下四个层次：

| 尺度层次 | 物理意义 | 关键参数 | 典型数值范围 |
|---------|---------|---------|-------------|
| **量子化学尺度（微观）** | 电子结构与分子轨道 | HOMO/LUMO 能级、能隙、偶极矩、极化率、结合能、电子亲和能、电离势 | HOMO: -15~-8 eV, LUMO: -3~2 eV, gap: 8~15 eV |
| **分子动力学尺度（介观）** | 分子构象与热力学行为 | 构象能量、旋转势垒、振动频率、熵、焓、自由能 | 构象能: -500~-200 kJ/mol |
| **状态方程尺度（宏观）** | 热力学性质与传输特性 | 沸点、临界温度/压力/密度、饱和蒸气压、密度、粘度、热导率、COP、体积制冷量 | 沸点: -80~50°C, COP: 2.5~6.0 |
| **环境与安全指标** | 环境影响与安全性能 | GWP(20yr/100yr)、ODP、大气寿命、可燃性等级、毒性分类、TFA 产率 | GWP: 0.1~14000, ODP: 0~1.0 |

### 1.3 实验流程总览

```
数据收集 → 数据清洗 → 特征工程 → 缺失值处理 → 特征标准化 → 
数据集划分 → 质量分析 → 输出数据集 → 生成模型训练
```

```python
    ])

    for mol_id, data in database.items():

                data['basic_info']['smiles'],

            )
            database[mol_id]['thermodynamics'].update(thermo_result)
    
    # ========== 步骤 5: 数据整合与验证 ==========
    validated_database = validate_data_consistency(database)
    
    return validated_database
```

### 3.3 数据增强策略

由于已知制冷剂分子数量有限 (~100 种)，需要通过数据增强扩充训练样本：

```python
def augment_dataset(df, n_augment_per_molecule=10):
    """
    数据增强：生成变体样本
    
    方法:
    1. 添加高斯噪声到连续特征
    2. 分子片段替换生成类似物
    3. 构象变化生成不同立体异构体
    """
    augmented_data = []
    
    for idx, row in df.iterrows():
        # 原始样本
        augmented_data.append(row)
        
        # 生成变体
        for i in range(n_augment_per_molecule):
            variant = row.copy()
            
            # 对数值特征添加小扰动 (5% 标准差)
            for col in numeric_columns:
                if pd.notna(row[col]) and row[col] != 0:
                    noise = np.random.normal(0, abs(row[col]) * 0.05)
                    variant[col] = row[col] + noise
            
            # 确保物理合理性约束
            variant = apply_physical_constraints(variant)
            
            variant['id'] = f"{row['id']}_aug{i}"
            augmented_data.append(variant)
    
    return pd.DataFrame(augmented_data)


def apply_physical_constraints(variant):
    """
    应用物理合理性约束
    
    确保增强后的数据符合物理规律
    """
    # 约束 1: GWP 必须为正
    variant['gwp_100yr'] = max(0.01, variant['gwp_100yr'])
    
    # 约束 2: COP 在合理范围内
    variant['cop'] = np.clip(variant['cop'], 1.5, 7.0)
    
    # 约束 3: 沸点在合理范围内
    variant['boiling_point_c'] = np.clip(variant['boiling_point_c'], -100, 100)
    
    # 约束 4: HOMO < LUMO
    if variant['homo_ev'] >= variant['lumo_ev']:
        variant['homo_ev'], variant['lumo_ev'] = variant['lumo_ev'] - 0.1, variant['homo_ev'] + 0.1
    
    # 约束 5: 能隙为正
    variant['gap_ev'] = max(1.0, variant['lumo_ev'] - variant['homo_ev'])
    
    return variant
```

### 4.2 数据清洗

#### 4.2.1 去重处理

```python
import pandas as pd

def remove_duplicates(df):
    """
    去除重复数据
    
    策略:
    1. 基于 SMILES 去重（分子结构唯一性）
    2. 基于 CAS 号去重（化学物质唯一性）
    3. 保留数据完整性最高的记录
    """
    initial_count = len(df)
    print(f"初始数据量：{initial_count}")
    
    # 策略 1: 基于 SMILES 去重
    # 对于相同 SMILES，保留缺失值最少的记录
    df['missing_count'] = df.isnull().sum(axis=1)
    df = df.sort_values('missing_count')
    df = df.drop_duplicates(subset=['smiles'], keep='first')
    
    # 策略 2: 基于 CAS 号去重
    df = df.drop_duplicates(subset=['cas_number'], keep='first')
    
    final_count = len(df)
    removed_count = initial_count - final_count
    print(f"去除重复：{removed_count} 条记录")
    print(f"最终数据量：{final_count}")
    
    # 删除临时列
    df = df.drop(columns=['missing_count'])
    
    return df
```

```python
def detect_and_handle_outliers(df, method='IQR', multiplier=3.0):
    """
    异常值检测与处理
    
    参数:
        df: 输入数据框
        method: 'IQR' 或 'ZScore'
        multiplier: IQR 倍数 (通常 1.5 为温和，3.0 为严格)
    
    返回:
        df_clean: 处理后的数据框
        outlier_report: 异常值报告
    """
    df_clean = df.copy()
    outlier_report = {}
    
    numeric_cols = df_clean.select_dtypes(include=[np.number]).columns
    
    for col in numeric_cols:
        data = df_clean[col].dropna()
        
        if method == 'IQR':
            Q1 = data.quantile(0.25)
            Q3 = data.quantile(0.75)
            IQR = Q3 - Q1
            
            lower_bound = Q1 - multiplier * IQR
            upper_bound = Q3 + multiplier * IQR
            
            outliers = (df_clean[col] < lower_bound) | (df_clean[col] > upper_bound)
            n_outliers = outliers.sum()
            
            if n_outliers > 0:
                outlier_report[col] = {
                    'n_outliers': int(n_outliers),
                    'lower_bound': lower_bound,
                    'upper_bound': upper_bound,
                    'outlier_values': df_clean.loc[outliers, col].tolist()
                }
                
                # Winsorization 处理：将异常值截断到边界
                df_clean[col] = df_clean[col].clip(lower_bound, upper_bound)
                print(f"列 {col}: 检测到 {n_outliers} 个异常值，已进行 Winsorization 处理")
        
        elif method == 'ZScore':
            z_scores = np.abs((data - data.mean()) / data.std())
            outliers = z_scores > 3.0
            
            n_outliers = outliers.sum()
            if n_outliers > 0:
                outlier_report[col] = {
                    'n_outliers': int(n_outliers),
                    'method': 'ZScore',
                    'threshold': 3.0
                }
                # 用中位数替换异常值
                median_val = data.median()
                df_clean.loc[outliers, col] = median_val
                print(f"列 {col}: 检测到 {n_outliers} 个异常值，已用中位数替换")
    
    return df_clean, outlier_report
```

```python
def validate_data_consistency(df):
    """
    数据一致性检查
    
    检查项目:
    1. GWP 与制冷剂类型的一致性
    2. ODP 值的合理性
    3. HOMO/LUMO 能级关系
    4. 热力学参数的物理合理性
    """
    df_valid = df.copy()
    consistency_issues = []
    
    # 检查 1: HFOs 应该具有低 GWP (< 10)
    hfo_mask = df_valid['type'] == 'HFO'
    high_gwp_hfo = df_valid[hfo_mask & (df_valid['gwp_100yr'] > 50)]
    if len(high_gwp_hfo) > 0:
        consistency_issues.append({
            'issue': 'HFO 高 GWP',
            'count': len(high_gwp_hfo),
            'ids': high_gwp_hfo['id'].tolist()
        })
        print(f"警告：{len(high_gwp_hfo)} 个 HFO 分子 GWP 值异常高 (>50)")
    
    # 检查 2: CFCs 应该有高 ODP (> 0.5)
    cfc_mask = df_valid['type'] == 'CFC'
    low_odp_cfc = df_valid[cfc_mask & (df_valid['odp'] < 0.3)]
    if len(low_odp_cfc) > 0:
        consistency_issues.append({
            'issue': 'CFC 低 ODP',
            'count': len(low_odp_cfc),
            'ids': low_odp_cfc['id'].tolist()
        })
        print(f"警告：{len(low_odp_cfc)} 个 CFC 分子 ODP 值异常低 (<0.3)")
    
    # 检查 3: HFCs/HFOs 的 ODP 应该为 0
    no_odp_mask = df_valid['type'].isin(['HFC', 'HFO', 'c-HFC'])
    non_zero_odp = df_valid[no_odp_mask & (df_valid['odp'] > 0.01)]
    if len(non_zero_odp) > 0:
        consistency_issues.append({
            'issue': '无 ODP 物质有 ODP 值',
            'count': len(non_zero_odp),
            'ids': non_zero_odp['id'].tolist()
        })
        # 修正：设为 0
        df_valid.loc[non_zero_odp.index, 'odp'] = 0.0
        print(f"修正：{len(non_zero_odp)} 个 HFC/HFO/c-HFC 分子的 ODP 设为 0")
    
    # 检查 4: HOMO 应该小于 LUMO
    invalid_gap = df_valid[df_valid['homo_ev'] >= df_valid['lumo_ev']]
    if len(invalid_gap) > 0:
        consistency_issues.append({
            'issue': 'HOMO >= LUMO',
            'count': len(invalid_gap),
            'ids': invalid_gap['id'].tolist()
        })
        # 交换 HOMO 和 LUMO
        for idx in invalid_gap.index:
            homo = df_valid.loc[idx, 'homo_ev']
            lumo = df_valid.loc[idx, 'lumo_ev']
            df_valid.loc[idx, 'homo_ev'] = min(homo, lumo) - 0.1
            df_valid.loc[idx, 'lumo_ev'] = max(homo, lumo) + 0.1
        print(f"修正：{len(invalid_gap)} 个分子的 HOMO/LUMO 关系")
    
    # 检查 5: 能隙应该为正
    negative_gap = df_valid[df_valid['gap_ev'] <= 0]
    if len(negative_gap) > 0:
        consistency_issues.append({
            'issue': '能隙 <= 0',
            'count': len(negative_gap),
            'ids': negative_gap['id'].tolist()
        })
        # 重新计算能隙
        df_valid['gap_ev'] = df_valid['lumo_ev'] - df_valid['homo_ev']
        df_valid['gap_ev'] = df_valid['gap_ev'].clip(1.0, None)  # 最小为 1 eV
        print(f"修正：{len(negative_gap)} 个分子的能隙")
    
    # 检查 6: 热力学参数合理性
    # 沸点范围：-100°C 到 100°C
    invalid_bp = df_valid[~df_valid['boiling_point_c'].between(-100, 100)]
    if len(invalid_bp) > 0:
        consistency_issues.append({
            'issue': '沸点超出合理范围',
            'count': len(invalid_bp),
            'ids': invalid_bp['id'].tolist()
        })
    
    # 检查 7: COP 范围：1.5 到 7.0
    invalid_cop = df_valid[~df_valid['cop'].between(1.5, 7.0)]
    if len(invalid_cop) > 0:
        consistency_issues.append({
            'issue': 'COP 超出合理范围',
            'count': len(invalid_cop),
            'ids': invalid_cop['id'].tolist()
        })
    
    # 添加一致性标记列
    df_valid['consistency_check'] = len(consistency_issues) == 0
    
    print(f"\n一致性检查完成，发现 {len(consistency_issues)} 类问题")
    
    return df_valid, consistency_issues
```

```python
                    'num_h_donors': np.nan,
                    'qed_score': np.nan,
                    'sa_score': np.nan
                }
            else:
                feat = {
                    # 基础描述符
                    'mol_weight': float(Descriptors.MolWt(mol)),
                    'logp': float(Descriptors.MolLogP(mol)),  # 疏水性
                    'tpsa': float(Descriptors.TPSA(mol)),      # 极性表面积
                    
                    # 拓扑描述符
                    'num_atoms': float(mol.GetNumAtoms()),
                    'num_rings': float(rdMolDescriptors.CalcNumRings(mol)),

                    # 合成可及性 (需要额外计算)
                    'sa_score': np.nan  # 需要额外计算
                }

            print(f"处理 SMILES {smiles} 时出错：{e}")
            descriptors.append({k: np.nan for k in feat.keys()})
    
    return pd.DataFrame(descriptors)
```

```python
def create_derived_features(df):

        df_feat['odp'] * 100 +          # ODP 权重较高
        df_feat['atmospheric_lifetime_yr'] / 50  # 归一化大气寿命
    )

        df_feat['atmospheric_lifetime_yr'] / 50
    )
    
    # ========== 3. 安全性评分 ==========
    # 将 ASHRAE 安全等级数值化
    safety_map = {
        'A1': 1,   # 无毒不燃
        'A2L': 2,  # 无毒弱燃
        'A2': 3,   # 无毒可燃
        'A3': 4,   # 无毒高燃
        'B1': 5,   # 有毒不燃
        'B2': 6,   # 有毒可燃
        'B3': 7    # 有毒高燃
    }
    df_feat['safety_score'] = df_feat['ashrae_safety'].map(safety_map)
    
    # 可燃性评分 (单独)
    flammability_map = {'A1': 1, 'A2L': 2, 'A2': 3, 'A3': 4}
    df_feat['flammability_score'] = df_feat['flammability_class'].map(flammability_map)
    
    # ========== 4. 综合性能指标 (多目标权衡) ==========
    # 综合 COP, GWP, 安全性的评分
    # 权重可根据具体应用调整
    df_feat['comprehensive_score'] = (
        df_feat['cop'] / 5.0 -           # COP 归一化 (目标值~5)
        df_feat['gwp_100yr'] / 5000 -    # GWP 归一化 (目标值<500)
        df_feat['safety_score'] / 7.0    # 安全性归一化
    )
    
    # 环境友好型评分 (仅考虑环境指标)
    df_feat['eco_friendly_score'] = (
        1.0 / (1 + df_feat['gwp_100yr'] / 1000) +
        (1 - df_feat['odp']) +

    )
```

```python
from sklearn.feature_selection import VarianceThreshold
from sklearn.decomposition import PCA

def select_important_features(df, target_col='comprehensive_score'):
    """
    特征选择
    
    方法:
    1. 方差阈值过滤
    2. 相关性过滤
    3. 基于模型的特征重要性
    """
    df_select = df.copy()
    
    # 仅选择数值型特征
    numeric_df = df_select.select_dtypes(include=[np.number])
    
    # 1. 方差阈值过滤 (去除低方差特征)
    var_threshold = VarianceThreshold(threshold=0.01)
    var_threshold.fit(numeric_df)
    high_var_cols = numeric_df.columns[var_threshold.get_support()]
    print(f"方差过滤后保留 {len(high_var_cols)} 个特征")
    
    # 2. 相关性过滤 (去除高度相关的特征)
    corr_matrix = numeric_df[high_var_cols].corr()
    
    # 找出高度相关的特征对 (|corr| > 0.95)
    highly_correlated_pairs = []
    for i in range(len(corr_matrix.columns)):
        for j in range(i):
            if abs(corr_matrix.iloc[i, j]) > 0.95:
                highly_correlated_pairs.append((corr_matrix.columns[i], corr_matrix.columns[j]))
    
    # 从每对中移除一个特征
    cols_to_remove = set()
    for col1, col2 in highly_correlated_pairs:
        if col1 not in cols_to_remove and col2 not in cols_to_remove:
            # 保留与目标变量相关性更高的特征
            corr_with_target1 = abs(numeric_df[col1].corr(numeric_df[target_col]))
            corr_with_target2 = abs(numeric_df[col2].corr(numeric_df[target_col]))
            if corr_with_target1 < corr_with_target2:
                cols_to_remove.add(col1)
            else:
                cols_to_remove.add(col2)
    
    selected_cols = [c for c in high_var_cols if c not in cols_to_remove]
    print(f"相关性过滤后保留 {len(selected_cols)} 个特征")
    
    # 3. 基于模型的特征重要性 (可选)
    from sklearn.ensemble import RandomForestRegressor
    
    X = numeric_df[selected_cols].dropna()
    y = numeric_df.loc[X.index, target_col].dropna()
    
    # 对齐索引
    common_idx = X.index.intersection(y.index)
    X = X.loc[common_idx]
    y = y.loc[common_idx]
    
    rf = RandomForestRegressor(n_estimators=100, random_state=42)
    rf.fit(X, y)
    
    # 获取特征重要性
    importance_df = pd.DataFrame({
        'feature': selected_cols,
        'importance': rf.feature_importances_
    }).sort_values('importance', ascending=False)
    
    # 选择重要性 > 0.01 的特征
    important_features = importance_df[importance_df['importance'] > 0.01]['feature'].tolist()
    print(f"基于模型选择后保留 {len(important_features)} 个特征")
    
    return important_features, importance_df
```

```python
from sklearn.impute import KNNImputer, SimpleImputer

def handle_missing_values(df, threshold=0.2):
    """
    缺失值处理
    
    多层次策略:
    1. 删除缺失率 > threshold 的列
    2. KNN 插补数值型特征
    3. 众数填充分类型特征
    """
    df_imp = df.copy()
    
    # ========== 步骤 1: 分析缺失情况 ==========
    missing_ratio = df_imp.isnull().sum() / len(df_imp)
    
    print("缺失率统计:")
    print(missing_ratio[missing_ratio > 0].sort_values(ascending=False))
    
    # ========== 步骤 2: 删除高缺失率列 ==========
    cols_to_drop = missing_ratio[missing_ratio > threshold].index.tolist()
    
    if cols_to_drop:
        print(f"\n删除缺失率 > {threshold*100:.0f}% 的列：{cols_to_drop}")
        df_imp = df_imp.drop(columns=cols_to_drop)
    
    # ========== 步骤 3: 数值型特征 KNN 插补 ==========
    numeric_cols = df_imp.select_dtypes(include=[np.number]).columns
    
    if len(numeric_cols) > 0:
        print(f"\n对 {len(numeric_cols)} 个数值型特征进行 KNN 插补...")
        
        imputer = KNNImputer(n_neighbors=5, weights='distance')
        df_imp[numeric_cols] = imputer.fit_transform(df_imp[numeric_cols])
        
        # 保存 imputer 以便后续使用
        # import joblib
        # joblib.dump(imputer, 'knn_imputer.pkl')
    
    # ========== 步骤 4: 分类型特征众数填充 ==========
    categorical_cols = df_imp.select_dtypes(include=['object']).columns
    
    for col in categorical_cols:
        if df_imp[col].isnull().any():
            mode_value = df_imp[col].mode()[0]
            df_imp[col] = df_imp[col].fillna(mode_value)
            print(f"列 {col}: 用众数 '{mode_value}' 填充 {df_imp[col].isnull().sum()} 个缺失值")
    
    # ========== 步骤 5: 验证缺失值处理结果 ==========
    remaining_missing = df_imp.isnull().sum().sum()
    print(f"\n缺失值处理完成，剩余缺失值：{remaining_missing}")
    
    return df_imp
```

```python
from sklearn.model_selection import train_test_split

def create_dataset_splits(df, test_size=0.2, val_size=0.15, stratify_col='type'):
    """
    创建训练集、验证集、测试集
    
    采用分层抽样确保各类别分子均匀分布
    
    参数:
        df: 完整数据集
        test_size: 测试集比例
        val_size: 验证集比例 (相对于剩余数据)
        stratify_col: 分层列名
    
    返回:
        splits: {'train': train_df, 'val': val_df, 'test': test_df}
    """
    # 检查分层列是否存在
    if stratify_col not in df.columns:
        print(f"警告：分层列 '{stratify_col}' 不存在，使用随机划分")
        stratify = None
    else:
        stratify = df[stratify_col]
    
    # 第一步：划分训练集和临时集
    train_df, temp_df = train_test_split(
        df,
        test_size=test_size,
        stratify=stratify,
        random_state=42
    )
    
    # 第二步：从临时集划分验证集和测试集
    val_size_adjusted = val_size / (1 - test_size)
    
    if stratify_col in temp_df.columns:
        stratify_temp = temp_df[stratify_col]
    else:
        stratify_temp = None
    
    val_df, test_df = train_test_split(
        temp_df,
        test_size=val_size_adjusted,
        stratify=stratify_temp,
        random_state=42
    )
    
    print(f"数据集划分完成:")
    print(f"  训练集：{len(train_df)} 样本 ({len(train_df)/len(df)*100:.1f}%)")
    print(f"  验证集：{len(val_df)} 样本 ({len(val_df)/len(df)*100:.1f}%)")
    print(f"  测试集：{len(test_df)} 样本 ({len(test_df)/len(df)*100:.1f}%)")
    
    # 验证分层效果
    if stratify_col:
        print(f"\n分层列 '{stratify_col}' 分布:")
        for split_name, split_df in [('训练集', train_df), ('验证集', val_df), ('测试集', test_df)]:
            dist = split_df[stratify_col].value_counts().to_dict()
            print(f"  {split_name}: {dist}")
    
    return {
        'train': train_df,
        'val': val_df,
        'test': test_df
    }
```

### 10.2 物理信息嵌入机制

基于论文 6.3 节"状态方程与量子计算融合"，在生成模型中嵌入多尺度物理信息：

```python
import torch
import torch.nn as nn

class PhysicsInformedEncoder(nn.Module):
    """
    物理信息编码器
    
    基于论文 6.3 节的物理信息嵌入机制
    将多尺度物理信息编码为隐空间表示
    """
    
    def __init__(self, quantum_dim=8, thermo_dim=10, env_dim=5, hidden_dim=256):
        super().__init__()
        
        # 量子化学信息编码器
        self.quantum_encoder = nn.Sequential(
            nn.Linear(quantum_dim, hidden_dim // 4),
            nn.ReLU(),
            nn.Linear(hidden_dim // 4, hidden_dim // 4)
        )
        
        # 热力学信息编码器
        self.thermo_encoder = nn.Sequential(
            nn.Linear(thermo_dim, hidden_dim // 4),
            nn.ReLU(),
            nn.Linear(hidden_dim // 4, hidden_dim // 4)
        )
        
        # 环境信息编码器
        self.env_encoder = nn.Sequential(
            nn.Linear(env_dim, hidden_dim // 4),
            nn.ReLU(),
            nn.Linear(hidden_dim // 4, hidden_dim // 4)
        )
        
        # 融合层
        self.fusion = nn.Sequential(
            nn.Linear(hidden_dim * 3 // 4, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim)
        )
    
    def forward(self, quantum_feat, thermo_feat, env_feat):
        """
        前向传播
        
        参数:
            quantum_feat: 量子化学特征 [batch, quantum_dim]
            thermo_feat: 热力学特征 [batch, thermo_dim]
            env_feat: 环境特征 [batch, env_dim]
        
        返回:
            physics_emb: 物理信息嵌入 [batch, hidden_dim]
        """
        # 分别编码
        q_emb = self.quantum_encoder(quantum_feat)
        t_emb = self.thermo_encoder(thermo_feat)
        e_emb = self.env_encoder(env_feat)
        
        # 拼接并融合
        combined = torch.cat([q_emb, t_emb, e_emb], dim=-1)
        physics_emb = self.fusion(combined)
        
        return physics_emb


class PhysicsInformedDiffusion(nn.Module):
    """
    物理信息嵌入的扩散模型
    
    基于论文 6.2 节"生成式模型架构"和 6.3 节"物理信息嵌入"
    """
    
    def __init__(self, molecule_dim=256, physics_dim=256, hidden_dim=512):
        super().__init__()
        
        # 物理信息编码器
        self.physics_encoder = PhysicsInformedEncoder(
            quantum_dim=8,
            thermo_dim=10,
            env_dim=5,
            hidden_dim=physics_dim
        )
        
        # 扩散模型主体
        self.diffusion_network = nn.Sequential(
            nn.Linear(molecule_dim + physics_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, molecule_dim)
        )
        
        # 时间嵌入
        self.time_embed = nn.Embedding(1000, physics_dim)
    
    def forward(self, x, t, quantum_feat, thermo_feat, env_feat):
        """
        前向传播
        
        参数:
            x: 分子表示 [batch, molecule_dim]
            t: 时间步 [batch]
            quantum_feat: 量子化学特征
            thermo_feat: 热力学特征
            env_feat: 环境特征
        """
        # 编码物理信息
        physics_emb = self.physics_encoder(quantum_feat, thermo_feat, env_feat)
        
        # 时间嵌入
        time_emb = self.time_embed(t)
        
        # 融合物理信息和时间信息
        condition_emb = physics_emb + time_emb
        
        # 条件扩散
        x_cond = torch.cat([x, condition_emb], dim=-1)
        output = self.diffusion_network(x_cond)
        
        return output
```

```python
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import r2_score, mean_absolute_error

def benchmark_data_quality(train_df, test_df, target_col='cop'):
    """
    使用性质预测模型验证数据质量
    
    目标：R² > 0.8 表示数据质量良好
    """
    # 选择特征
    feature_cols = [c for c in train_df.select_dtypes(include=[np.number]).columns 
                   if c != target_col and c not in ['id']]
    
    X_train = train_df[feature_cols].dropna()
    y_train = train_df.loc[X_train.index, target_col]
    
    X_test = test_df[feature_cols].dropna()
    y_test = test_df.loc[X_test.index, target_col]
    
    # 对齐索引
    common_cols = list(set(X_train.columns) & set(X_test.columns))
    X_train = X_train[common_cols]
    X_test = X_test[common_cols]
    
    # 训练模型
    model = RandomForestRegressor(n_estimators=100, random_state=42)
    model.fit(X_train, y_train)
    
    # 预测
    y_pred = model.predict(X_test)
    
    # 评估
    r2 = r2_score(y_test, y_pred)
    mae = mean_absolute_error(y_test, y_pred)
    
    print(f"基准模型性能 (目标：{target_col}):")
    print(f"  R² = {r2:.3f} (目标：>0.8)")
    print(f"  MAE = {mae:.3f}")
    
    if r2 > 0.8:
        print("✓ 数据质量良好，适合用于生成模型训练")
    else:
        print("⚠ 数据质量可能需要改进")
    
    return r2, mae
```
